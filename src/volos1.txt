#include <TFT_eSPI.h> 

TFT_eSPI tft = TFT_eSPI();
TFT_eSprite back = TFT_eSprite(&tft);
TFT_eSprite needle = TFT_eSprite(&tft);

void setup() {
  tft.init();
  tft.fillScreen(TFT_BLACK);
  // by default pivot point is center of item
 // tft.setPivot(tft.width()/2,tft.width()/2);
  tft.setPivot(100,100);  // has no effect
  needle.createSprite(20,80);
  //needle.setPivot(15,60);
  back.createSprite(tft.width()-2,tft.width()-2);
//  back.setPivot(back.width()/2,back.width()/2);
  //back.setPivot(100,100);
}

int angle=0;

void loop() {
  back.fillSprite(TFT_BLACK);
  back.fillCircle(tft.width()/2,tft.width()/2,tft.width()/2,TFT_SILVER);
  needle.fillSprite(TFT_BLACK);
  needle.drawWedgeLine(10,0,10,40,1,10,TFT_RED);
  needle.drawWedgeLine(10,40,10,80,10,1,TFT_BLUE);
  needle.fillCircle(10,40,10,TFT_WHITE);
  needle.pushRotated(&back,angle,TFT_BLACK);
  back.pushSprite(1,1);
  angle++;
  if(angle==360)
    angle=0;

}
